# 学習データセット回復手順

```sql
CREATE TABLE IF NOT EXISTS systems (
    id          INTEGER PRIMARY KEY,
    name        VARCHAR(128) NOT NULL
);

CREATE TABLE IF NOT EXISTS readings (
    system_id   INTEGER NOT NULL,
    read_on     TIMESTAMP NOT NULL,
    power       DECIMAL(10,3) NOT NULL
            DEFAULT 0 CHECK(power >= 0), -- デフォルト値: 0 CHECKにて0以上
    PRIMARY KEY (system_id, read_on),
    FOREIGN KEY (system_id)
            REFERENCES systems(id)
);

INSERT INTO systems(id, name)
         SELECT DISTINCT system_id, system_public_name
         FROM 'https://oedi-data-lake.s3.amazonaws.com/pvdaq/csv/systems.csv'
         ORDER BY system_id ASC;

INSERT INTO readings(system_id, read_on, power)
         SELECT system_id, measured_on,
             CASE
                 WHEN ac_power_hw__2695 < 0 OR ac_power_hw__2695 IS NULL THEN 0
                 ELSE ac_power_hw__2695 END AS ac_power
         FROM read_csv_auto(
           's3://oedi-data-lake/pvdaq/csv/pvdata/system_id=34/year=2019/*/*/*.csv',
           union_by_name=True);

CREATE OR REPLACE VIEW v_power_per_day AS
SELECT system_id,
    date_trunc('day', read_on) AS day,
    round(sum(power) / 4 / 1000, 2) AS kWh,
FROM readings
GROUP BY system_id, day;

SELECT count(system_id) FROM readings;
```

### 式としてのサブクエリ

式としてのサブクエリは、常に1行だけ返す必要がある。

`EXISTS` 別のテーブルの1行に存在する値を持つ行をすべて選択する。

```sql
.mode line
-- vという名前の列を1つ持つインラインテーブル sを定義
SELECT * FROM VALUES (7), (11) s(v)
WHERE EXISTS (SELECT * FROM range(10) WHERE range = v);
v = 7
```

`IN` 少なくとも1回含まれている場合、真と評価される。

```sql
.mode line
SELECT * FROM VALUES (7), (11) s(v)
WHERE v IN (SELECT * FROM range(10));
```

`ANY` 外側の値と内側の値いずれかとの比較が真として評価される。

```sql
.mode line
SELECT * FROM VALUES (7), (11) s(v)
WHERE v = ANY (SELECT * FROM range(10));
```

`ALL` 外側の値と内側の値との比較がすべて真として評価される。

```sql
.mode line
SELECT * FROM VALUES (7), (11) s(v)
WHERE v = ALL (SELECT 7);
```

## グループ化セット

様々な集約を使う。

```sql
.mode duckbox
SELECT count(*),
  min(power) AS min_W, max(power) AS max_W,
  -- 15分毎のW単位からkWhに変換
  round(sum(power) / 4 / 1000, 2) AS kWh
FROM readings;
```

```sql
-- kWhでシステム毎の発電量を求める
SELECT
  year(read_on) AS year,
  system_id, count(*),
  round(sum(power) / 4 / 1000, 2) AS kWh
FROM readings
GROUP BY year, system_id
ORDER BY year, system_id;
```

`GROUPING SETS`

```sql
-- 測定値と総発電量それぞれの
-- 年毎、年とシステム毎、全体の値を求める
SELECT
  year(read_on) AS year,
  system_id, count(*),
  round(sum(power) / 4 / 1000, 2) AS kWh
FROM readings
GROUP BY GROUPING SETS ((year, system_id), year, ())
ORDER BY year NULLS FIRST, system_id NULLS FIRST;
```

`GROUP BY GROUPING SETS ((year, system_id), year, ())` の意味

- `(year, system_id)` `year` \* `system_id` の組み合わせパケット
- `year` のみ、このパケットに含まれないキーはNULLをあてる
- `()` 空のパケットまたは空のグループ

`GROUP BY ROLLUP` を使って書き換える。

```sql
SELECT
  year(read_on) AS year,
  system_id, count(*),
  round(sum(power) / 4 / 1000, 2) AS kWh
FROM readings
GROUP BY ROLLUP (year, system_id)
ORDER BY year NULLS FIRST, system_id NULLS FIRST;
```

`GROUP BY CUBE`

```sql
-- GROUP BY CUBE を使ってシステム毎の合計を求める
SELECT
  year(read_on) AS year,
  system_id, count(*),
  round(sum(power) / 4 / 1000, 2) AS kWh
FROM readings
GROUP BY CUBE (year, system_id)
ORDER BY year NULLS FIRST, system_id NULLS FIRST;
```

## ウィンドウ関数

- 他の行を見ることができるようになる
- `GROUP BY` のように集約されない
- `OVER` 句を用いて宣言する

```sql
-- NG: 最も多くの電力が生産された上位3つの時刻
SELECT * FROM readings ORDER BY power DESC LIMIT 3;
```

```sql
-- OK: 最も多くの電力が生産された上位3つの時刻
WITH ranked_readings AS (
  SELECT *,
    dense_rank()
    -- ウィンドウは発電量の多い順で1行毎に開かれる
    OVER (ORDER BY power DESC) AS rnk
  FROM readings
)
SELECT *
FROM ranked_readings
WHERE rnk <= 3;
```

### パーティションの定義

`PARTITION BY` `COLUMN_NAME` を引数に与えてウィンドウとして機能する。

```sql
-- システム毎の消費電力量上位2位
WITH ranked_readings AS (
  SELECT *,
  dense_rank()
  OVER (-- 移動するウィンドウの定義開始
    PARTITION BY system_id -- パーティション定義、ウィンドウの自体
    ORDER BY power DESC
  ) AS rnk
  FROM readings
)
SELECT * FROM ranked_readings WHERE rnk <= 2
ORDER BY system_id, rnk ASC;
```

```sql
-- システム毎の毎日の発電量と
-- システム全体の平均発電量
SELECT *,
  avg(kWh)
    OVER (
      PARTITION BY system_id
    ) AS average_per_system
FROM v_power_per_day LIMIT 3;
```

### フレーミング

関数が評価される各行から見た相対的な行集合を指定する。現在の行からの距離は、行の前/行の後に指定された式として与えられる。

```sql
SELECT system_id, day, kWh, avg(kWh)
  OVER (
    PARTITION BY system_id -- system_id毎にパーティション上を移動
    ORDER BY day ASC -- 日付昇順
    RANGE BETWEEN INTERVAL 3 Days PRECEDING  - - 前に3日間,合計7日間
      AND INTERVAL 3 Days FOLLOWING -- 後に3日間
  ) AS "kWh 7-day moving average"
FROM v_power_per_day
ORDER BY system_id, day;
```

### 名前付きウィンドウ

**名前付きウィンドウ** 複数の集約を求めるクエリの場合に利用する。

```sql
-- 外れ値を除外し分位数を求める
-- 集約(min,max,quantiles)を使う
SELECT system_id,
  day,
  min(kWh) OVER seven_days AS "7-day min", -- 定義されたウィンドウをFROM句で参照
  quantile(kWh, [0.25, 0.5, 0.75]) -- 分位数を計算
    OVER seven_days AS "kWh 7-day quantile",
  max(kWh) OVER seven_days AS "7-day max",
FROM v_power_per_day
WINDOW -- FROM語句の後で宣言
  seven_days AS (
    PARTITION BY system_id, month(day)
    ORDER BY day ASC
    RANGE BETWEEN INTERVAL 3 Days PRECEDING
      AND INTERVAL 3 Days FOLLOWING
  )
ORDER BY system_id, day;
```

### パーティション内行アクセス

```sql
-- 元の価格と新しい価格の差
SELECT valid_from,
  value, -- value列を取り出す
  lag(value)
    OVER validity AS "Previous value",
  -- 現在の行の価格とその前の行の価格の差
  value - lag(value, 1, value)
    OVER validity AS Change
FROM prices
WHERE date_part('year', valid_from) = 2019
WINDOW validity AS (ORDER BY valid_from)
ORDER BY valid_from;
```

集約関数の中にウィンドウ関数の呼び出しをネストできない

```sql
-- 2019年の価格の総変動
-- 呼び出しをネストできないのでCTEを使う
WITH changes AS (
  SELECT value - lag(value, 1, value) OVER (ORDER BY valid_from) AS v
  FROM prices
  WHERE date_part('year', valid_from) = 2019
  ORDER BY valid_from
)
SELECT sum(changes.v) AS total_change
FROM changes;
```

## WHERE句の外の条件とフィルタリング

集約の計算結果やウィンドウ関数の結果は `WHERE` 句でフィルタリングできない

- **HAVING** 句 : 驟雨訳された値が特定の値Xを超えるグループの選択で利用する
- **QUALIFY** 句 : 一定期間において特定の値を超える選択で利用する

フィルタリングのための句と使用方法

| 句      | 使う場所                                                                                         | 効果                                                         |
| :------ | :----------------------------------------------------------------------------------------------- | :----------------------------------------------------------- |
| HAVING  | `GROUP BY` 句の後                                                                                | グループに対して計算された集約値に基づいて行をフィルタリング |
| QUALIFY | ウィンドウ式を参照する `FROM` 句の後、そのウィンドウで計算された内容に基づいて行をフィルタリング |
| FILTER  | 集約関数の後                                                                                     | その集約に渡される値をフィルタリング                         |

### HAVING句の使用

```sql
SELECT system_id,
  date_trunc('day', read_on) AS day,
  round(sum(power) / 4 / 1000, 2) AS kWh
FROM readings
WHERE kWh >= 900
GROUP BY ALL;

Binder Error:
WHERE clause cannot contain aggregates!

LINE 3:   round(sum(power) / 4 / 1000, 2) AS kWh
```

`WHERE` 句に集約を含めることができない。計算で求められる `kWh` 列は `WHERE` 句が適用された時点では計算されていない。

```sql
-- 発電量が900kWhを超える日
SELECT system_id,
  date_trunc('day', read_on) AS day,
  round(sum(power) / 4 / 1000, 2) AS kWh
FROM readings
GROUP BY ALL
HAVING kWh >= 900
ORDER BY kWh DESC;
```

### QUALIFY句の使用

ウィンドウ関数の結果に基づいてフィルタリングする。

```sql
-- QUALIFY句でウィンドウ内の集約値をフィルタリング
SELECT
  dense_rank() OVER (ORDER BY power DESC) AS rnk,
FROM readings
QUALIFY rnk <= 3;
```

```sql
-- 7日間のウィンドウ内平均が850を超えた日

) AS "kWh 7-day moving average"
FROM v_power_per_day
QUALIFY "kWh 7-day moving average" > 850
```

### FILTER句の使用

行から除外する。

```sql
INSERT INTO readings(system_id, read_on, power)
SELECT any_value(SiteId),
  time_bucket(
    INTERVAL '15 Minutes',
    CAST("Date-Time" AS timestamp)
  ) AS read_on,
  coalesce(avg(ac_power)
  FILTER (
    ac_power IS NOT NULL AND
    ac_power >= 0

  read_csv_auto(

ORDER BY read_on
```

```sql
-- 年毎の総発電量と最高発電日を求める
PIVOT v_power_per_day
```
