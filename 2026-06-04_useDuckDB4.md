`time_bucket()`
: タイムスタンプを指定した間隔に切り捨て、任意のオフセットに整列させる。

```sql
INSERT INTO readings(system_id, read_on, power)
SELECT any_value(SiteId),
  time_bucket(
    INTERVAL '15 Minutes',
    CAST("Date-Time" AS timestamp)
  ) AS read_on,
avg(
  CASE
    WHEN ac_power < 0 OR ac_power IS NULL THEN 0
    ELSE ac_power END)
FROM
  read_csv_auto(
    's3://oedi-data-lake/pvdaq/csv/pvdata/system_id=34/year=2019/*/*/*.csv'
  )
GROUP BY read_on
ORDER BY read_on;

-- レコードレイアウトがかわったため動作せず
```

## サブクエリ

```sql
-- 非相関サブクエリ
-- 生産した合計電力の平均
-- sum()/avg() を利用
SELECT avg(sum_per_system)
FROM (
  SELECT sum(kWh) AS sum_per_system
  FROM v_power_per_day
  GROUP BY system_id
);
┌─────────────────────┐
│ avg(sum_per_system) │
│       double        │
├─────────────────────┤
│  102871.73999999998 │
└─────────────────────┘
```

```sql
-- スカラ非相関サブクエリ
-- どの日にどのシステムで最も多くの電力を生産したのか
SELECT read_on, power
FROM readings
WHERE power = (
  SELECT max(power) FROM readings
);
```

```sql
-- スカラ相関サブクエリ
SELECT system_id, read_on, power
FROM readings r1
WHERE power = (
  SELECT max(power)
  FROM readings r2
  -- 外側クエリと内側サブクエリを結合
  WHERE r2.system_id = r1.system_id
)
ORDER BY ALL;
┌───────────┬─────────────────────┬───────────────┐
│ system_id │       read_on       │     power     │
```

外部テーブルとの非相関サブクエリ結合

```sql
SELECT r1.system_id, read_on, power
FROM readings r1
JOIN (
  SELECT r2.system_id, max(power) AS value
  FROM readings r2
  GROUP BY ALL
) AS max_power ON (
  max_power.system_id = r1.system_id AND
  max_power.value = r1.power
)
ORDER BY ALL;
┌───────────┬─────────────────────┬───────────────┐
│ system_id │       read_on       │     power     │
│   int32   │      timestamp      │ decimal(10,3) │
```
